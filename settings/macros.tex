	\begingroup%
	\def\chapterautorefname{Chapter}%
	\def\sectionautorefname{Section}%

	\def\figureautorefname{Figure}%
	\def\tableautorefname{Table}%
	\def\lstlistingautorefname{Listing}%
	\autoref{#1}%
	\endgroup%
}
